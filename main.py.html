import pygame
from sys import exit

pygame.init()
screen = pygame.display.set_mode((800, 400))
pygame.display.set_caption('Rustin')
clock = pygame.time.Clock()
test_font = pygame.font.Font(r"C:\Users\Jacob\Desktop\graphics\ARCADECLASSIC.ttf", 50)

sky_surface = pygame.image.load(r'C:\Users\Jacob\Desktop\graphics\skybox1.png')
ground_surface = pygame.image.load(r'C:\Users\Jacob\Desktop\graphics\ground3.png')
box_surface = pygame.image.load(r'C:\Users\Jacob\Desktop\graphics\box.png').convert_alpha()
tree_surface = pygame.image.load(r'C:\Users\Jacob\Desktop\graphics\tree.png').convert_alpha()
text_surface = test_font.render('GAMING', False, '#00008B')

box_x_pos = 100
box_rect = box_surface.get_rect(midbottom = (80,320))
tree_rect = tree_surface.get_rect(midbottom = (400, 315))

while True:
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
            exit()

    dest1 = (0, 0)
    screen.blit(sky_surface,dest1)

    dest2 = (0, 300)
    screen.blit(ground_surface, dest2)

    dest3 = (300, 350)
    screen.blit(text_surface, dest3)




    screen.blit(box_surface, box_rect)

    box_rect.left += 1

    screen.blit(tree_surface, tree_rect)


    if box_x_pos == 799:
        box_x_pos = 1


    print(box_rect.colliderect(tree_rect))



    #draw all our elements
    # update everything
    pygame.display.update()
    clock.tick(60)
